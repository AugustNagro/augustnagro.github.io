  50 /**
  51  * Implementations of {@link Collector} that implement various useful reduction
  52  * operations, such as accumulating elements into collections, summarizing
  53  * elements according to various criteria, etc.
  54  *
  55  * <p>The following are examples of using the predefined collectors to perform
  56  * common mutable reduction tasks:
  57  *
  58  * <pre>{@code
  59  * // Accumulate names into a List
  60  * List<String> list = people.stream()
  61  *   .map(Person::getName)
  62  *   .collect(Collectors.toList());
  63  *
  64  * // Accumulate names into a TreeSet
  65  * Set<String> set = people.stream()
  66  *   .map(Person::getName)
  67  *   .collect(Collectors.toCollection(TreeSet::new));
  68  *
  69  * // Convert elements to strings and concatenate them, separated by commas
  70  * String joined = things.stream()
  71  *   .map(Object::toString)
  72  *   .collect(Collectors.joining(", "));
  73  *
  74  * // Compute sum of salaries of employee
  75  * int total = employees.stream()
  76  *   .collect(Collectors.summingInt(Employee::getSalary));
  77  *
  78  * // Group employees by department
  79  * Map<Department, List<Employee>> byDept = employees.stream()
  80  *   .collect(Collectors.groupingBy(Employee::getDepartment));
  81  *
  82  * // Compute sum of salaries by department
  83  * Map<Department, Integer> totalByDept = employees.stream()
  84  *   .collect(Collectors.groupingBy(Employee::getDepartment,
  85  *                                  Collectors.summingInt(Employee::getSalary)));
  86  *
  87  * // Partition students into passing and failing
  88  * Map<Boolean, List<Student>> passingFailing = students.stream()
  89  *   .collect(Collectors.partitioningBy(s -> s.getGrade() >= PASS_THRESHOLD));
  90  *
  91  * }</pre>
  92  *
  93  * @since 1.8
  94  */

 129     /**
 130      * {@code BinaryOperator<Map>} that merges the contents of its right
 131      * argument into its left argument, throwing {@code IllegalStateException}
 132      * if duplicate keys are encountered.
 133      *
 134      * @param <K> type of the map keys
 135      * @param <V> type of the map values
 136      * @param <M> type of the map
 137      * @return a merge function for two maps
 138      */
 139     private static <K, V, M extends Map<K,V>>
 140     BinaryOperator<M> uniqKeysMapMerger() {
 141         return (m1, m2) -> {
 142             for (Map.Entry<K,V> e : m2.entrySet()) {
 143                 K k = e.getKey();
 144                 V v = Objects.requireNonNull(e.getValue());
 145                 V u = m1.putIfAbsent(k, v);
 146                 if (u != null) throw duplicateKeyException(k, u, v);
 147             }
 148             return m1;
 149         };
 150     }

 152     /**
 153      * {@code BiConsumer<Map, T>} that accumulates (key, value) pairs
 154      * extracted from elements into the map, throwing {@code IllegalStateException}
 155      * if duplicate keys are encountered.
 156      *
 157      * @param keyMapper a function that maps an element into a key
 158      * @param valueMapper a function that maps an element into a value
 159      * @param <T> type of elements
 160      * @param <K> type of map keys
 161      * @param <V> type of map values
 162      * @return an accumulating consumer
 163      */
 164     private static <T, K, V>
 165     BiConsumer<Map<K, V>, T> uniqKeysMapAccumulator(Function<? super T, ? extends K> keyMapper,
 166                                                     Function<? super T, ? extends V> valueMapper) {
 167         return (map, element) -> {
 168             K k = keyMapper.apply(element);
 169             V v = Objects.requireNonNull(valueMapper.apply(element));
 170             V u = map.putIfAbsent(k, v);
 171             if (u != null) throw duplicateKeyException(k, u, v);
 172         };
 173     }

 486                                    downstream.characteristics());

 489     /**
 490      * Adapts a {@code Collector} accepting elements of type {@code U} to one
 491      * accepting elements of type {@code T} by applying a flat mapping function
 492      * to each input element before accumulation.  The flat mapping function
 493      * maps an input element to a {@link Stream stream} covering zero or more
 494      * output elements that are then accumulated downstream.  Each mapped stream
 495      * is {@link java.util.stream.BaseStream#close() closed} after its contents
 496      * have been placed downstream.  (If a mapped stream is {@code null}
 497      * an empty stream is used, instead.)
 498      *
 499      * @apiNote
 500      * The {@code flatMapping()} collectors are most useful when used in a
 501      * multi-level reduction, such as downstream of a {@code groupingBy} or
 502      * {@code partitioningBy}.  For example, given a stream of
 503      * {@code Order}, to accumulate the set of line items for each customer:
 504      * <pre>{@code
 505      * Map<String, Set<LineItem>> itemsByCustomerName
 506      *   = orders.stream().collect(
 507      *     groupingBy(Order::getCustomerName,
 508      *                flatMapping(order -> order.getLineItems().stream(),
 509      *                            toSet())));
 510      * }</pre>
 511      *
 512      * @param <T> the type of the input elements
 513      * @param <U> type of elements accepted by downstream collector
 514      * @param <A> intermediate accumulation type of the downstream collector
 515      * @param <R> result type of collector
 516      * @param mapper a function to be applied to the input elements, which
 517      * returns a stream of results
 518      * @param downstream a collector which will receive the elements of the
 519      * stream returned by mapper
 520      * @return a collector which applies the mapping function to the input
 521      * elements and provides the flat mapped results to the downstream collector
 522      * @since 9
 523      */
 524     public static <T, U, A, R>
 525     Collector<T, ?, R> flatMapping(Function<? super T, ? extends Stream<? extends U>> mapper,
 526                                    Collector<? super U, A, R> downstream) {
 527         BiConsumer<A, ? super U> downstreamAccumulator = downstream.accumulator();
 528         return new CollectorImpl<>(downstream.sizedSupplier(), downstream.supplier(),
 529                             (r, t) -> {
 530                                 try (Stream<? extends U> result = mapper.apply(t)) {
 531                                     if (result != null)
 532                                         result.sequential().forEach(u -> downstreamAccumulator.accept(r, u));
 533                                 }
 534                             },
 535                             downstream.combiner(), downstream.finisher(),
 536                             downstream.characteristics());
 537     }

 539     /**
 540      * Adapts a {@code Collector} to one accepting elements of the same type
 541      * {@code T} by applying the predicate to each input element and only
 542      * accumulating if the predicate returns {@code true}.
 543      *
 544      * @apiNote
 545      * The {@code filtering()} collectors are most useful when used in a
 546      * multi-level reduction, such as downstream of a {@code groupingBy} or
 547      * {@code partitioningBy}.  For example, given a stream of
 548      * {@code Employee}, to accumulate the employees in each department that have a
 549      * salary above a certain threshold:
 550      * <pre>{@code
 551      * Map<Department, Set<Employee>> wellPaidEmployeesByDepartment
 552      *   = employees.stream().collect(
 553      *     groupingBy(Employee::getDepartment,
 554      *                filtering(e -> e.getSalary() > 2000,
 555      *                          toSet())));
 556      * }</pre>

 558      * In this example, suppose there are no employees whose salary is above the
 559      * threshold in some department.  Using a filtering collector as shown above
 560      * would result in a mapping from that department to an empty {@code Set}.
 561      * If a stream {@code filter()} operation were done instead, there would be
 562      * no mapping for that department at all.
 563      *
 564      * @param <T> the type of the input elements
 565      * @param <A> intermediate accumulation type of the downstream collector
 566      * @param <R> result type of collector
 567      * @param predicate a predicate to be applied to the input elements
 568      * @param downstream a collector which will accept values that match the
 569      * predicate
 570      * @return a collector which applies the predicate to the input elements
 571      * and provides matching elements to the downstream collector
 572      * @since 9
 573      */

 575     Collector<T, ?, R> filtering(Predicate<? super T> predicate,
 576                                  Collector<? super T, A, R> downstream) {
 577         BiConsumer<A, ? super T> downstreamAccumulator = downstream.accumulator();
 578         return new CollectorImpl<>(downstream.sizedSupplier(), downstream.supplier(),
 579                                    (r, t) -> {
 580                                        if (predicate.test(t)) {
 581                                            downstreamAccumulator.accept(r, t);
 582                                        }
 583                                    },
 584                                    downstream.combiner(), downstream.finisher(),
 585                                    downstream.characteristics());
 586     }

 720     /**
 721      * Returns a {@code Collector} that produces the sum of a double-valued
 722      * function applied to the input elements.  If no elements are present,
 723      * the result is 0.
 724      *
 725      * <p>The sum returned can vary depending upon the order in which
 726      * values are recorded, due to accumulated rounding error in
 727      * addition of values of differing magnitudes. Values sorted by increasing
 728      * absolute magnitude tend to yield more accurate results.  If any recorded
 729      * value is a {@code NaN} or the sum is at any point a {@code NaN} then the
 730      * sum will be {@code NaN}.
 731      *
 732      * @param <T> the type of the input elements
 733      * @param mapper a function extracting the property to be summed
 734      * @return a {@code Collector} that produces the sum of a derived property
 735      */
 736     public static <T> Collector<T, ?, Double>
 737     summingDouble(ToDoubleFunction<? super T> mapper) {
 738         /*
 739          * In the arrays allocated for the collect operation, index 0
 740          * holds the high-order bits of the running sum, index 1 holds
 741          * the low-order bits of the sum computed via compensated
 742          * summation, and index 2 holds the simple sum used to compute
 743          * the proper result if the stream contains infinite values of
 744          * the same sign.
 745          */
 746         return new CollectorImpl<>(
 747                 () -> new double[3],
 748                 (a, t) -> { double val = mapper.applyAsDouble(t);
 749                             sumWithCompensation(a, val);
 750                             a[2] += val;},
 751                 (a, b) -> { sumWithCompensation(a, b[0]);
 752                             a[2] += b[2];
 753                             return sumWithCompensation(a, b[1]); },
 754                 a -> computeFinalSum(a),
 755                 CH_NOID);
 756     }

 758     /**
 759      * Incorporate a new double value using Kahan summation /
 760      * compensation summation.
 761      *
 762      * High-order bits of the sum are in intermediateSum[0], low-order
 763      * bits of the sum are in intermediateSum[1], any additional
 764      * elements are application-specific.
 765      *
 766      * @param intermediateSum the high-order and low-order words of the intermediate sum
 767      * @param value the name value to be included in the running sum
 768      */
 769     static double[] sumWithCompensation(double[] intermediateSum, double value) {
 770         double tmp = value - intermediateSum[1];
 771         double sum = intermediateSum[0];
 772         double velvel = sum + tmp; // Little wolf of rounding error
 773         intermediateSum[1] = (velvel - sum) - tmp;
 774         intermediateSum[0] = velvel;
 775         return intermediateSum;
 776     }

 778     /**
 779      * If the compensated sum is spuriously NaN from accumulating one
 780      * or more same-signed infinite values, return the
 781      * correctly-signed infinity stored in the simple sum.
 782      */
 783     static double computeFinalSum(double[] summands) {
 784         // Better error bounds to add both terms as the final sum
 785         double tmp = summands[0] + summands[1];
 786         double simpleSum = summands[summands.length - 1];
 787         if (Double.isNaN(tmp) && Double.isInfinite(simpleSum))
 788             return simpleSum;
 789         else
 790             return tmp;
 791     }

1300      * @see #groupingBy(Function, Supplier, Collector)
1301      */

1303     Collector<T, ?, M> groupingByConcurrent(Function<? super T, ? extends K> classifier,
1304                                             Supplier<M> mapFactory,
1305                                             Collector<? super T, A, D> downstream) {
1306         Supplier<A> downstreamSupplier = downstream.supplier();
1307         BiConsumer<A, ? super T> downstreamAccumulator = downstream.accumulator();
1308         BinaryOperator<ConcurrentMap<K, A>> merger = Collectors.<K, A, ConcurrentMap<K, A>>mapMerger(downstream.combiner());
1309         @SuppressWarnings("unchecked")
1310         Supplier<ConcurrentMap<K, A>> mangledFactory = (Supplier<ConcurrentMap<K, A>>) mapFactory;
1311         BiConsumer<ConcurrentMap<K, A>, T> accumulator;

1313             accumulator = (m, t) -> {
1314                 K key = Objects.requireNonNull(classifier.apply(t), "element cannot be mapped to a null key");

1319         else {

1321                 K key = Objects.requireNonNull(classifier.apply(t), "element cannot be mapped to a null key");
1322                 A resultContainer = m.computeIfAbsent(key, k -> downstreamSupplier.get());
1323                 synchronized (resultContainer) {
1324                     downstreamAccumulator.accept(resultContainer, t);
1325                 }
1326             };
1327         }
1328 
1329         if (downstream.characteristics().contains(Collector.Characteristics.IDENTITY_FINISH)) {
1330             return new CollectorImpl<>(mangledFactory, accumulator, merger, CH_CONCURRENT_ID);
1331         }
1332         else {
1333             @SuppressWarnings("unchecked")
1334             Function<A, A> downstreamFinisher = (Function<A, A>) downstream.finisher();
1335             Function<ConcurrentMap<K, A>, M> finisher = intermediate -> {
1336                 intermediate.replaceAll((k, v) -> downstreamFinisher.apply(v));
1337                 @SuppressWarnings("unchecked")
1338                 M castResult = (M) intermediate;
1339                 return castResult;
1340             };

1391      * @param <D> the result type of the downstream reduction

1411                 new Partition<>(downstream.supplier().get(),

1417             Function<Partition<A>, Map<Boolean, D>> finisher = par ->
1418                     new Partition<>(downstream.finisher().apply(par.forTrue),
1419                                     downstream.finisher().apply(par.forFalse));

1917      * into the final result.

1919      * <p>The resulting collector functions do the following:
1920      *

1940      * @param downstream2 the second downstream collector
1941      * @param merger      the function which merges two results into the single one
1942      * @return a {@code Collector} which aggregates the results of two supplied collectors.

1952     private static <T, A1, A2, R1, R2, R>
1953     Collector<T, ?, R> teeing0(Collector<? super T, A1, R1> downstream1,
1954                                Collector<? super T, A2, R2> downstream2,
1955                                BiFunction<? super R1, ? super R2, R> merger) {
1956         Objects.requireNonNull(downstream1, "downstream1");
1957         Objects.requireNonNull(downstream2, "downstream2");
1958         Objects.requireNonNull(merger, "merger");
1959 


1960         BiConsumer<A1, ? super T> c1Accumulator =
1961                 Objects.requireNonNull(downstream1.accumulator(), "downstream1 accumulator");
1962         BiConsumer<A2, ? super T> c2Accumulator =
1963                 Objects.requireNonNull(downstream2.accumulator(), "downstream2 accumulator");
1964         BinaryOperator<A1> c1Combiner = Objects.requireNonNull(downstream1.combiner(), "downstream1 combiner");
1965         BinaryOperator<A2> c2Combiner = Objects.requireNonNull(downstream2.combiner(), "downstream2 combiner");
1966         Function<A1, R1> c1Finisher = Objects.requireNonNull(downstream1.finisher(), "downstream1 finisher");
1967         Function<A2, R2> c2Finisher = Objects.requireNonNull(downstream2.finisher(), "downstream2 finisher");
1968 
1969         Set<Collector.Characteristics> characteristics;
1970         Set<Collector.Characteristics> c1Characteristics = downstream1.characteristics();
1971         Set<Collector.Characteristics> c2Characteristics = downstream2.characteristics();
1972         if (CH_ID.containsAll(c1Characteristics) || CH_ID.containsAll(c2Characteristics)) {
1973             characteristics = CH_NOID;
1974         } else {
1975             EnumSet<Collector.Characteristics> c = EnumSet.noneOf(Collector.Characteristics.class);
1976             c.addAll(c1Characteristics);
1977             c.retainAll(c2Characteristics);
1978             c.remove(Collector.Characteristics.IDENTITY_FINISH);
1979             characteristics = Collections.unmodifiableSet(c);
1980         }
1981 
1982         class PairBox {
1983             A1 left;
1984             A2 right;
1985 
1986             PairBox(int initialSize) {
1987                 left = Objects.requireNonNull(downstream1.sizedSupplier(), "downstream1 sizedSupplier")
1988                         .apply(initialSize);
1989                 right = Objects.requireNonNull(downstream2.sizedSupplier(), "downstream2 sizedSupplier")
1990                         .apply(initialSize);
1991             }
1992 
1993             PairBox() {
1994                 left = Objects.requireNonNull(downstream1.supplier(), "downstream1 supplier").get();
1995                 right = Objects.requireNonNull(downstream2.supplier(), "downstream2 supplier").get();
1996             }
1997 
1998             void add(T t) {
1999                 c1Accumulator.accept(left, t);
2000                 c2Accumulator.accept(right, t);
2001             }
2002 
2003             PairBox combine(PairBox other) {
2004                 left = c1Combiner.apply(left, other.left);
2005                 right = c2Combiner.apply(right, other.right);
2006                 return this;
2007             }
2008 
2009             R get() {
2010                 R1 r1 = c1Finisher.apply(left);
2011                 R2 r2 = c2Finisher.apply(right);
2012                 return merger.apply(r1, r2);
2013             }
2014         }
2015 
2016         return new CollectorImpl<>(PairBox::new,
2017                                    PairBox::new,
2018                                    PairBox::add,
2019                                    PairBox::combine,
2020                                    PairBox::get,
2021                                    characteristics);
2022     }
